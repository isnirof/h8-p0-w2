//2. Tahun Kabisat

SET tahun WITH VALUE 1995

IF (tahun MODULO 4 EQUAL TO 0) AND (tahun MODULO 100 NOT EQUAL TO 0)
    DISPLAY "Tahun Kabisat"

ELSE IF (tahun MODULO 4 EQUAL TO 0) AND (tahun MODULO 100 EQUAL TO 0) AND (tahun MODULO 400 EQUAL TO 0)
    DIPLAY "Tahun Kabisat"

ELSE
    DISPLAY "Bukan Tahun Kabisat"
